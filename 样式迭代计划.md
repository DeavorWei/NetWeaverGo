## 二、动效系统 🎬

### 2.1 动画令牌化

创建 `tokens/animations.css`：

```css
:root {
  /* 动画时长 */
  --duration-instant: 100ms;
  --duration-fast: 150ms;
  --duration-normal: 250ms;
  --duration-slow: 400ms;

  /* 缓动函数 */
  --ease-default: cubic-bezier(0.4, 0, 0.2, 1);
  --ease-in: cubic-bezier(0.4, 0, 1, 1);
  --ease-out: cubic-bezier(0, 0, 0.2, 1);
  --ease-bounce: cubic-bezier(0.68, -0.55, 0.265, 1.55);

  /* 预设动画 */
  --animate-fade-in: fadeIn var(--duration-normal) var(--ease-out);
  --animate-slide-up: slideUp var(--duration-normal) var(--ease-out);
  --animate-scale-in: scaleIn var(--duration-fast) var(--ease-bounce);
}
```

### 2.2 动画工具类

```css
/* utilities/animations.css */
.animate-fade {
  animation: var(--animate-fade-in);
}
.animate-slide-up {
  animation: var(--animate-slide-up);
}
.animate-pulse {
  animation: pulse 2s infinite;
}
.animate-spin {
  animation: spin 1s linear infinite;
}
```

## 四、响应式系统 📱

### 4.1 断点令牌化

创建 `tokens/breakpoints.css`：

```css
:root {
  --breakpoint-sm: 640px; /* 手机横屏 */
  --breakpoint-md: 768px; /* 平板竖屏 */
  --breakpoint-lg: 1024px; /* 平板横屏/小笔记本 */
  --breakpoint-xl: 1280px; /* 桌面 */
  --breakpoint-2xl: 1536px; /* 大屏 */
}
```

### 4.2 响应式工具类

```css
/* 移动端优先 */

  @media (min-width: 640px) {
    display: flex;
  }
}
.md:grid-cols-2 {
  @media (min-width: 768px) {
    grid-template-columns: repeat(2, 1fr);
  }
}
```

## 五、无障碍访问 (Accessibility) ♿

### 5.1 焦点可见性增强

```css
/* 焦点环样式 */
.focus-visible-ring:focus-visible {
  outline: 2px solid var(--color-accent);
  outline-offset: 2px;
}

/* 高对比度焦点 */
@media (forced-colors: active) {
  :focus-visible {
    outline: 3px solid CanvasText;
  }
}
```

### 5.2 减少动画偏好支持

```css
@media (prefers-reduced-motion: reduce) {
  *,
  *::before,
  *::after {
    animation-duration: 0.01ms !important;
    animation-iteration-count: 1 !important;
    transition-duration: 0.01ms !important;
  }
}
```

### 5.3 高对比度媒体查询

```css
@media (prefers-contrast: high) {
  :root {
    --color-border-default: #000000;
    --color-text-primary: #000000;
  }
}
```

## 六、性能优化 ⚡

### 6.1 CSS 按需加载

```typescript
// 动态加载主题
const loadTheme = async (theme: string) => {
  const link = document.createElement("link");
  link.rel = "stylesheet";
  link.href = `/themes/${theme}.css`;
  document.head.appendChild(link);
};
```
