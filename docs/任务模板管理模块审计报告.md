## 2. 问题清单

### 问题 #1: 编辑保护存在TOCTOU竞态条件 [高危]

**严重程度**: 🔴 高危  
**问题类型**: 并发安全  
**影响范围**: [`UpdateTaskGroup`](internal/ui/task_group_service.go:161), [`DeleteTaskGroup`](internal/ui/task_group_service.go:181)

**问题描述**:

编辑保护机制存在经典的"检查-使用"时间差（Time-Of-Check to Time-Of-Use, TOCTOU）竞态条件。当前实现流程为：

1. [`canEditTaskGroup`](internal/ui/task_group_service.go:196) 检查是否存在活跃运行
2. 如果检查通过，执行更新/删除操作
3. 在步骤1和步骤2之间存在时间窗口，其他协程可能启动新任务

**问题代码**:

```go
// internal/ui/task_group_service.go:161-178
func (s *TaskGroupService) UpdateTaskGroup(id uint, group models.TaskGroup) (*models.TaskGroup, error) {
    existing, err := config.GetTaskGroup(id)
    if err != nil {
        return nil, err
    }

    if !s.canEditTaskGroup(existing.ID) {  // 检查点
        return nil, fmt.Errorf("任务存在活跃运行，当前不可编辑")
    }

    // ⚠️ 时间窗口：此时可能有新任务启动
    
    group.ID = existing.ID
    // ... 执行更新
    return config.UpdateTaskGroup(id, group)  // 使用点
}
```

**影响分析**:
- 可能导致正在执行的任务模板被修改，影响任务执行一致性
- 可能导致任务执行过程中模板被删除，引发运行时错误
- 数据完整性风险

## 3. 修复方案

### 3.1 问题 #1 修复方案：引入乐观锁或互斥保护

**方案A：数据库乐观锁（推荐）**

在 `task_groups` 表添加 `version` 字段，利用GORM的乐观锁机制：

```go
// internal/models/models.go
type TaskGroup struct {
    ID        uint      `json:"id" gorm:"primaryKey;autoIncrement"`
    Version   int       `json:"version" gorm:"version"` // 乐观锁版本号
    // ... 其他字段
}

// internal/ui/task_group_service.go
func (s *TaskGroupService) UpdateTaskGroup(id uint, group models.TaskGroup) (*models.TaskGroup, error) {
    // 使用事务包装检查和更新
    tx := config.DB.Begin()
    defer func() {
        if r := recover(); r != nil {
            tx.Rollback()
        }
    }()
    
    var existing models.TaskGroup
    if err := tx.First(&existing, id).Error; err != nil {
        tx.Rollback()
        return nil, err
    }
    
    // 在事务内检查活跃运行
    if s.hasActiveRunInTx(tx, existing.ID) {
        tx.Rollback()
        return nil, fmt.Errorf("任务存在活跃运行，当前不可编辑")
    }
    
    group.ID = existing.ID
    group.Version = existing.Version + 1
    // ... 其他字段
    
    result := tx.Model(&models.TaskGroup{}).Where("id = ? AND version = ?", id, existing.Version).Updates(group)
    if result.Error != nil {
        tx.Rollback()
        return nil, result.Error
    }
    if result.RowsAffected == 0 {
        tx.Rollback()
        return nil, fmt.Errorf("数据已被修改，请刷新后重试")
    }
    
    tx.Commit()
    return &group, nil
}
```

```go
func (s *TaskGroupService) queryRunMetadata() (map[uint]*taskexec.RunSummary, map[uint]int) {
    result := make(map[uint]*taskexec.RunSummary)
    activeCount := make(map[uint]int)
    
    if s.taskexec == nil {
        return result, activeCount
    }

    runs, err := s.taskexec.ListRuns(500)
    if err != nil {
        return result, activeCount
    }

    for _, run := range runs {
        if run == nil || run.TaskGroupID == 0 {
            continue
        }
        
        // 记录最新运行
        if _, exists := result[run.TaskGroupID]; !exists {
            result[run.TaskGroupID] = run
        }
        
        // 统计活跃运行
        if taskexec.IsActiveRunStatus(strings.TrimSpace(run.Status)) {
            activeCount[run.TaskGroupID]++
        }
    }
    
    return result, activeCount
}
```

### 3.4 问题 #4 修复方案：改进拓扑字段验证

```go
// internal/config/task_group.go
func validateTaskGroup(group *models.TaskGroup) error {
    // ... 前置验证
    
    if strings.TrimSpace(group.TaskType) != "topology" {
        return nil
    }

    catalog := make(map[string]models.TopologyFieldSpec)
    for _, spec := range topologyFieldCatalogForValidation() {
        fieldKey := strings.TrimSpace(spec.FieldKey)
        if fieldKey == "" {
            continue
        }
        catalog[fieldKey] = spec
    }
    
    // 改进：提供更明确的错误信息
    if len(catalog) == 0 {
        return fmt.Errorf("系统配置错误：拓扑字段目录未初始化，请联系管理员")
    }

    // ... 后续验证逻辑
    
    // 改进：明确Enabled的默认语义
    for _, item := range group.TopologyFieldOverrides {
        fieldKey := strings.TrimSpace(item.FieldKey)
        if fieldKey == "" {
            continue
        }
        
        spec, ok := catalog[fieldKey]
        if !ok {
            return fmt.Errorf("拓扑字段覆盖包含无效字段: %s", fieldKey)
        }
        
        // 明确语义：Enabled为nil时使用默认值
        enabled := spec.DefaultEnabled
        if item.Enabled != nil {
            enabled = *item.Enabled
        }
        
        if enabled && strings.TrimSpace(item.Command) == "" {
            return fmt.Errorf("拓扑字段 %s 已启用但命令为空", fieldKey)
        }
    }
    
    return nil
}
```

```typescript
// 统一加载方法

  if (selectedTaskType.value !== "topology") {
    return;
  }
  
  topologyState.value.loading = true;
  topologyState.value.error = "";
  
  try {
    const nextPreview = await TopologyCommandAPI.previewTopologyCommands(
      topologyVendor.value,
      selectedDevices.value.map((item) => item.id),
      cloneTopologyOverrides(topologyState.value.overrides),
    );

  } finally {
    topologyState.value.loading = false;
  }
}
```
