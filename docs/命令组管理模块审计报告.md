#### 🔴 问题01: 导入功能路径遍历漏洞

| 属性 | 值 |
|------|-----|
| **严重程度** | 高危 |
| **问题位置** | [`ImportCommandGroup()`](internal/config/command_group.go:161) |
| **影响范围** | 系统安全 |
| **CVE参考** | CWE-22 Path Traversal |

**问题描述**:  
[`ImportCommandGroup()`](internal/config/command_group.go:161) 函数直接使用用户传入的 `filePath` 参数调用 `os.ReadFile()`，未进行路径验证和规范化处理。攻击者可以通过传入 `../../../etc/passwd` 等路径读取系统任意文件。

**漏洞代码**:
```go
// internal/config/command_group.go:166-168
data, err := os.ReadFile(filePath)
if err != nil {
    return nil, fmt.Errorf("读取导入文件失败: %v", err)
}
```

**攻击场景**:
1. 攻击者构造恶意请求：`ImportCommandGroup("../../../etc/passwd")`
2. 系统读取敏感文件内容并尝试解析为JSON
3. 虽然JSON解析会失败，但错误信息可能泄露文件存在性

### 2.2 并发安全问题

#### 🔴 问题05: 名称唯一性检查存在竞态条件

| 属性 | 值 |
|------|-----|
| **严重程度** | 高危 |
| **问题位置** | [`CreateCommandGroup()`](internal/config/command_group.go:56), [`UpdateCommandGroup()`](internal/config/command_group.go:81), [`DuplicateCommandGroup()`](internal/config/command_group.go:126), [`ImportCommandGroup()`](internal/config/command_group.go:161) |
| **影响范围** | 数据一致性 |
| **CVE参考** | CWE-362 Race Condition |

**问题描述**:  
所有涉及名称唯一性检查的函数都存在"检查-然后-操作"(Check-Then-Act)的竞态条件。在并发请求下，多个请求可能同时通过名称检查，导致创建重复名称的命令组。

**漏洞代码**:
```go
// internal/config/command_group.go:66-73
var count int64
DB.Model(&models.CommandGroup{}).Where("name = ?", group.Name).Count(&count)
if count > 0 {
    return nil, fmt.Errorf("命令组名称已存在: %s", group.Name)
}
// ⚠️ 时间窗口：此时其他请求可能已创建同名命令组
if err := DB.Create(&group).Error; err != nil {
    return nil, err
}
```

**修复方案**:
```go
// 方案1: 使用数据库唯一约束 + 错误处理
func CreateCommandGroup(group models.CommandGroup) (*models.CommandGroup, error) {
    group.Name = strings.TrimSpace(group.Name)
    if group.Name == "" {
        return nil, fmt.Errorf("命令组名称不能为空")
    }

    if err := DB.Create(&group).Error; err != nil {
        if errors.Is(err, gorm.ErrDuplicatedKey) {
            return nil, fmt.Errorf("命令组名称已存在: %s", group.Name)
        }
        return nil, err
    }

    logger.Info("Config", "-", "成功创建命令组: %s", group.Name)
    return &group, nil
}

// 方案2: 使用事务 + 行级锁
func CreateCommandGroup(group models.CommandGroup) (*models.CommandGroup, error) {
    err := DB.Transaction(func(tx *gorm.DB) error {
        // SELECT FOR UPDATE 锁定检查
        var existing models.CommandGroup
        if err := tx.Set("gorm:query_option", "FOR UPDATE").
            Where("name = ?", group.Name).
            First(&existing).Error; err == nil {
            return fmt.Errorf("命令组名称已存在: %s", group.Name)
        } else if !errors.Is(err, gorm.ErrRecordNotFound) {
            return err
        }
        return tx.Create(&group).Error
    })
    // ...
}
```

**修复方案**:
```go
// 优化方案：一次性查询所有同名前缀的命令组
func generateUniqueName(baseName string) string {
    var existingNames []string
    DB.Model(&models.CommandGroup{}).
        Where("name LIKE ?", baseName+"%").
        Pluck("name", &existingNames)
    
    nameSet := make(map[string]bool)
    for _, name := range existingNames {
        nameSet[name] = true
    }
    
    if !nameSet[baseName] {
        return baseName
    }
    
    for i := 1; i <= len(existingNames)+1; i++ {
        newName := fmt.Sprintf("%s (%d)", baseName, i)
        if !nameSet[newName] {
            return newName
        }
    }
    return baseName
}
```

#### 修复问题01-02: 路径遍历漏洞

在 [`internal/config/command_group.go`](internal/config/command_group.go) 中添加路径验证函数：

```go
// internal/config/command_group.go

import (
    "path/filepath"
    "strings"
)

// 允许的导入导出目录（可配置）
var allowedImportDirs = []string{
    GetPathManager().GetStorageRoot(),
}

// validateFilePath 验证文件路径安全性
func validateFilePath(filePath string, allowCreate bool) (string, error) {
    // 获取绝对路径
    absPath, err := filepath.Abs(filePath)
    if err != nil {
        return "", fmt.Errorf("无效的文件路径")
    }
    
    // 检查路径遍历攻击
    cleanPath := filepath.Clean(absPath)
    if strings.Contains(cleanPath, "..") {
        return "", fmt.Errorf("路径不能包含上级目录引用")
    }
    
    // 检查是否在允许的目录内
    allowed := false
    for _, dir := range allowedImportDirs {
        if strings.HasPrefix(cleanPath, dir) {
            allowed = true
            break
        }
    }
    if !allowed {
        return "", fmt.Errorf("文件路径必须在允许的目录内")
    }
    
    // 检查文件扩展名
    ext := filepath.Ext(cleanPath)
    if ext != ".json" {
        return "", fmt.Errorf("只支持 JSON 格式文件")
    }
    
    return cleanPath, nil
}

// ImportCommandGroup 修复后的导入函数
func ImportCommandGroup(filePath string) (*models.CommandGroup, error) {
    if DB == nil {
        return nil, fmt.Errorf("数据库未初始化")
    }
    
    // 验证路径安全性
    safePath, err := validateFilePath(filePath, false)
    if err != nil {
        return nil, err
    }
    
    // 检查文件大小
    info, err := os.Stat(safePath)
    if err != nil {
        return nil, fmt.Errorf("读取导入文件失败")
    }
    if info.Size() > 10<<20 { // 10MB
        return nil, fmt.Errorf("导入文件过大，最大支持10MB")
    }
    
    data, err := os.ReadFile(safePath)
    if err != nil {
        logger.Error("Config", "-", "读取导入文件失败: %s, err: %v", safePath, err)
        return nil, fmt.Errorf("读取导入文件失败")
    }
    
    var group models.CommandGroup
    if err := json.Unmarshal(data, &group); err != nil {
        return nil, fmt.Errorf("解析导入文件失败，请检查JSON格式")
    }
    
    // 清除ID和时间戳
    group.ID = 0
    group.CreatedAt = time.Time{}
    group.UpdatedAt = time.Time{}
    
    // 验证名称
    group.Name = strings.TrimSpace(group.Name)
    if group.Name == "" {
        return nil, fmt.Errorf("命令组名称不能为空")
    }
    
    // 验证命令内容
    for i, cmd := range group.Commands {
        cmd = strings.TrimSpace(cmd)
        if len(cmd) > 4096 {
            return nil, fmt.Errorf("命令 %d 过长", i+1)
        }
        group.Commands[i] = cmd
    }
    
    // 使用事务创建（处理并发）
    err = DB.Transaction(func(tx *gorm.DB) error {
        // 生成唯一名称
        group.Name = generateUniqueNameTx(tx, group.Name)
        return tx.Create(&group).Error
    })
    if err != nil {
        return nil, err
    }
    
    logger.Info("Config", "-", "成功导入命令组: %s", group.Name)
    return &group, nil
}

// ExportCommandGroup 修复后的导出函数
func ExportCommandGroup(id uint, filePath string) error {
    if DB == nil {
        return fmt.Errorf("数据库未初始化")
    }
    
    // 验证路径安全性
    safePath, err := validateFilePath(filePath, true)
    if err != nil {
        return err
    }
    
    var group models.CommandGroup
    if err := DB.First(&group, id).Error; err != nil {
        return fmt.Errorf("未找到命令组: %d", id)
    }
    
    jsonData, err := json.MarshalIndent(group, "", "  ")
    if err != nil {
        return fmt.Errorf("序列化命令组失败")
    }
    
    // 确保目录存在
    dir := filepath.Dir(safePath)
    if err := os.MkdirAll(dir, 0755); err != nil {
        return fmt.Errorf("创建目录失败")
    }
    
    // 使用安全权限写入文件
    if err := os.WriteFile(safePath, jsonData, 0644); err != nil {
        logger.Error("Config", "-", "写入导出文件失败: %s, err: %v", safePath, err)
        return fmt.Errorf("写入导出文件失败")
    }
    
    logger.Info("Config", "-", "成功导出命令组到: %s", safePath)
    return nil
}
```

#### 修复问题05: 竞态条件

在数据库模型中添加唯一约束（已存在），并修改创建/更新逻辑：

```go
// internal/config/command_group.go

import (
    "errors"
    "gorm.io/gorm"
)

func CreateCommandGroup(group models.CommandGroup) (*models.CommandGroup, error) {
    if DB == nil {
        return nil, fmt.Errorf("数据库未初始化")
    }

    group.Name = strings.TrimSpace(group.Name)
    if group.Name == "" {
        return nil, fmt.Errorf("命令组名称不能为空")
    }

    // 直接尝试创建，依赖数据库唯一约束
    if err := DB.Create(&group).Error; err != nil {
        if errors.Is(err, gorm.ErrDuplicatedKey) {
            return nil, fmt.Errorf("命令组名称已存在: %s", group.Name)
        }
        return nil, err
    }

    logger.Info("Config", "-", "成功创建命令组: %s", group.Name)
    return &group, nil
}

// generateUniqueNameTx 在事务内生成唯一名称
func generateUniqueNameTx(tx *gorm.DB, baseName string) string {
    var existingNames []string
    tx.Model(&models.CommandGroup{}).
        Where("name LIKE ?", baseName+"%").
        Pluck("name", &existingNames)
    
    nameSet := make(map[string]bool)
    for _, name := range existingNames {
        nameSet[name] = true
    }
    
    if !nameSet[baseName] {
        return baseName
    }
    
    for i := 1; ; i++ {
        newName := fmt.Sprintf("%s (%d)", baseName, i)
        if !nameSet[newName] {
            return newName
        }
    }
}
```
