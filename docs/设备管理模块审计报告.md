### 2.1 严重问题（Critical）

#### 问题 C-01：设备密码明文存储

| 属性 | 描述 |
|------|------|
| **严重程度** | 🔴 严重 (Critical) |
| **问题描述** | [`DeviceAsset`](internal/models/models.go:12) 模型中的 `Password` 字段直接以明文形式存储在数据库中，未进行加密处理 |
| **影响范围** | 所有设备凭据数据 |
| **风险分析** | 数据库泄露将直接暴露所有设备的登录密码，攻击者可利用这些凭据接管网络设备 |

**代码位置**:
```go
// internal/models/models.go:17
Password    string    `json:"password" gorm:"column:password"`
```

**对比分析**:  
SNMP 凭据模块已实现 AES-256-GCM 加密（见 [`internal/snmp/crypto.go`](internal/snmp/crypto.go)），但设备管理模块未采用相同的安全措施。

**修复方案**:
```go
// 1. 在 internal/models/models.go 中添加密码加密标记
type DeviceAsset struct {
    // ...
    Password    string    `json:"password" gorm:"column:password"`
    // 添加内部字段用于判断是否已加密
    passwordEncrypted bool `gorm:"-"`
}

// 2. 在 device_service.go 的 AddDevice/UpdateDevice 中加密
func (s *DeviceService) AddDevice(device models.DeviceAsset) error {
    if device.Password != "" {
        encrypted, err := s.crypto.EncryptCredential(device.Password)
        if err != nil {
            return fmt.Errorf("加密密码失败: %w", err)
        }
        device.Password = encrypted
    }
    // ... 原有逻辑
}

// 3. 在 GetDeviceByID 中解密
func (s *DeviceService) GetDeviceByID(id uint) (*models.DeviceAssetResponse, error) {
    device, err := s.repo.FindByID(id)
    if err != nil {
        return nil, err
    }
    // 解密密码供编辑使用
    if device.Password != "" && IsEncrypted(device.Password) {
        decrypted, err := s.crypto.DecryptCredential(device.Password)
        if err != nil {
            return nil, fmt.Errorf("解密密码失败: %w", err)
        }
        device.Password = decrypted
    }
    return device.ToResponse(), nil
}
```

#### 问题 C-02：IP 冲突检查存在 TOCTOU 竞态条件

| 属性 | 描述 |
|------|------|
| **严重程度** | 🔴 严重 (Critical) |
| **问题描述** | [`AddDevices`](internal/ui/device_service.go:111) 方法中 IP 存在性检查与实际插入操作之间存在时间窗口，可能导致重复 IP 被写入 |
| **影响范围** | 批量添加设备场景 |
| **风险分析** | 并发请求可能导致唯一性约束失效，产生数据不一致 |

**代码位置**:
```go
// internal/ui/device_service.go:157-168
existing, err := s.repo.FindByIPs(ips)  // 检查点
if err != nil {
    return err
}
if len(existing) > 0 {
    return fmt.Errorf("IP 地址 %s 已存在", existing[0].IP)
}
// ⚠️ 时间窗口：此处可能被其他请求插入相同 IP
return s.repo.CreateBatch(expandedDevices)  // 使用点
```

**修复方案**:
```go
// 方案1：使用数据库唯一约束 + 事务回滚
func (s *DeviceService) AddDevices(devices []models.DeviceAsset) error {
    // ... 展开和校验逻辑
    
    err := s.repo.Transaction(func(tx repository.DeviceRepository) error {
        return tx.CreateBatch(expandedDevices)
    })
    
    if err != nil {
        // 检查是否为唯一约束冲突
        if strings.Contains(err.Error(), "UNIQUE constraint failed") ||
           strings.Contains(err.Error(), "duplicate key") {
            return fmt.Errorf("存在重复的 IP 地址")
        }
        return err
    }
    return nil
}

// 方案2：使用分布式锁（适用于分布式部署）
func (s *DeviceService) AddDevices(devices []models.DeviceAsset) error {
    // 获取设备 IP 范围的分布式锁
    lock := s.lockManager.AcquireLock("device:ip:batch", 30*time.Second)
    defer lock.Release()
    
    // ... 原有逻辑
}
```

// 定义业务错误类型
type BusinessError struct {
    Code    string
    Message string
    Detail  string // 仅用于日志
}

func (e *BusinessError) Error() string {
    return e.Message
}

func (e *BusinessError) Unwrap() error {
    return nil
}

// 使用示例
var (

func (s *DeviceService) GetDeviceByID(id uint) (*models.DeviceAssetResponse, error) {

### 2.3 中危问题（Medium）

#### 问题 M-01：批量更新存在 N+1 查询问题

| 属性 | 描述 |
|------|------|
| **严重程度** | 🟡 中危 (Medium) |
| **问题描述** | [`UpdateDevices`](internal/ui/device_service.go:243) 在循环中逐个查询设备信息进行密码合并 |
| **影响范围** | 批量更新性能 |
| **风险分析** | 大批量更新时将产生大量数据库查询，影响系统性能 |

**代码位置**:
```go
// internal/ui/device_service.go:266-273
for i := range devices {
    existing, err := s.repo.FindByID(devices[i].ID)  // N 次查询
    if err != nil {
        return fmt.Errorf("未找到设备: %d", devices[i].ID)
    }
    pwdResult := config.MergePassword(existing.Password, devices[i].Password)
    devices[i].Password = pwdResult.Password
}
```

**修复方案**:
```go
func (s *DeviceService) UpdateDevices(devices []models.DeviceAsset) error {
    if len(devices) == 0 {
        return nil
    }

    // 校验阶段
    for i := range devices {
        config.NormalizeDevice(&devices[i])
        if err := config.ValidateDevice(&devices[i]); err != nil {
            return fmt.Errorf("第 %d 台设备: %v", i+1, err)
        }
        if devices[i].ID == 0 {
            return fmt.Errorf("批量更新时存在无效设备 ID")
        }
    }

    // 批量查询现有设备
    ids := make([]uint, 0, len(devices))
    for _, d := range devices {
        ids = append(ids, d.ID)
    }
    
    existingDevices, err := s.repo.FindByIDs(ids)  // 单次批量查询
    if err != nil {
        return err
    }
    
    // 构建 ID -> Device 映射
    existingMap := make(map[uint]*models.DeviceAsset, len(existingDevices))
    for i := range existingDevices {
        existingMap[existingDevices[i].ID] = &existingDevices[i]
    }
    
    // 密码合并
    for i := range devices {
        existing, ok := existingMap[devices[i].ID]
        if !ok {
            return fmt.Errorf("未找到设备: %d", devices[i].ID)
        }
        pwdResult := config.MergePassword(existing.Password, devices[i].Password)
        devices[i].Password = pwdResult.Password
    }

    return s.repo.UpdateBatch(devices)
}
```

**修复方案**:
```typescript
// 使用 SecurePassword 类处理密码
class SecurePassword {
    private buffer: Uint8Array;
    private length: number = 0;
    
    constructor(password: string) {
        this.buffer = new TextEncoder().encode(password);
        this.length = this.buffer.length;
    }
    
    getValue(): string {
        return new TextDecoder().decode(this.buffer.slice(0, this.length));
    }
    
    clear(): void {
        this.buffer.fill(0);
        this.length = 0;
    }
}

// 在组件中使用
const securePassword = ref<SecurePassword | null>(null);

function closeModal() {
    showModal.value = false;
    if (securePassword.value) {
        securePassword.value.clear();
        securePassword.value = null;
    }
    formData.value = undefined;
}
```

#### 问题 M-03：缺少输入长度限制

| 属性 | 描述 |

| **影响范围** | 设备新增、编辑 |

**代码位置**:
```go
// internal/config/config.go:99-120
func ValidateDevice(device *models.DeviceAsset) error {

        return fmt.Errorf("IP 地址不能为空")
    }
    // ... 缺少长度校验

```

**修复方案**:
```go
const (
    MaxIPLength          = 45    // IPv6 最大长度
    MaxUsernameLength    = 64
    MaxPasswordLength    = 256
    MaxGroupLength       = 128
    MaxDisplayNameLength = 256
    MaxVendorLength      = 64
    MaxRoleLength        = 64
    MaxSiteLength        = 128
    MaxDescriptionLength = 1024
    MaxTagLength         = 64
    MaxTagsCount         = 32
)

func ValidateDevice(device *models.DeviceAsset) error {
    if device.IP == "" {
        return fmt.Errorf("IP 地址不能为空")
    }
    if len(device.IP) > MaxIPLength {
        return fmt.Errorf("IP 地址长度不能超过 %d 个字符", MaxIPLength)
    }
    if len(device.Username) > MaxUsernameLength {
        return fmt.Errorf("用户名长度不能超过 %d 个字符", MaxUsernameLength)
    }
    if len(device.Password) > MaxPasswordLength {
        return fmt.Errorf("密码长度不能超过 %d 个字符", MaxPasswordLength)
    }
    if len(device.Group) > MaxGroupLength {
        return fmt.Errorf("分组名称长度不能超过 %d 个字符", MaxGroupLength)
    }
    if len(device.DisplayName) > MaxDisplayNameLength {
        return fmt.Errorf("显示名称长度不能超过 %d 个字符", MaxDisplayNameLength)
    }
    if len(device.Vendor) > MaxVendorLength {
        return fmt.Errorf("厂商名称长度不能超过 %d 个字符", MaxVendorLength)
    }
    if len(device.Role) > MaxRoleLength {
        return fmt.Errorf("角色名称长度不能超过 %d 个字符", MaxRoleLength)
    }
    if len(device.Site) > MaxSiteLength {
        return fmt.Errorf("站点名称长度不能超过 %d 个字符", MaxSiteLength)
    }
    if len(device.Description) > MaxDescriptionLength {
        return fmt.Errorf("描述长度不能超过 %d 个字符", MaxDescriptionLength)
    }
    if len(device.Tags) > MaxTagsCount {
        return fmt.Errorf("标签数量不能超过 %d 个", MaxTagsCount)
    }
    for _, tag := range device.Tags {
        if len(tag) > MaxTagLength {
            return fmt.Errorf("单个标签长度不能超过 %d 个字符", MaxTagLength)
        }
    }
    
    // 原有校验逻辑
    ip := net.ParseIP(device.IP)
    if ip == nil {
        return fmt.Errorf("IP 地址格式不正确: %s", device.IP)
    }
    if device.Port <= 0 || device.Port > 65535 {
        return fmt.Errorf("端口号必须在 1-65535 之间")
    }
    if !isValidProtocol(device.Protocol) {
        return fmt.Errorf("无效的协议类型: %s", device.Protocol)
    }
    
    return nil
}
```

### 3.2 风险评估

设备管理模块存在以下主要安全风险：

1. **凭据安全风险**：密码明文存储是最严重的问题，一旦数据库泄露，所有设备凭据将直接暴露。建议优先修复问题 C-01。

2. **数据一致性风险**：并发场景下的竞态条件可能导致数据不一致，建议在数据库层面添加唯一约束并使用事务保护。

3. **类型安全风险**：前端大量使用 `as unknown as` 类型断言绕过类型检查，可能导致运行时错误。

### 3.3 修复优先级建议

| 优先级 | 问题编号 | 修复建议 |
|--------|----------|----------|
| P0 | C-01 | 立即实现密码加密存储，参考 SNMP 模块的 AES-256-GCM 实现 |
| P0 | C-02 | 添加数据库唯一约束 + 事务保护 |
| P1 | H-01 | 实现真正的批量更新接口 |
